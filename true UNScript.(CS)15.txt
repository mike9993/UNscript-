-- STREAMING_CHUNK:Setting up auto-save configuration...
local CFG_FOLDER = "UNScript"
local CFG_FILE   = "settings"
local CFG_EXT    = ".cfg"

local function cfgSafely(fn, ...)
if fn then
local ok, res = pcall(fn, ...)
if not ok then return nil end
return res
end
end

local function ensureCfgFolder()
if isfolder and not cfgSafely(isfolder, CFG_FOLDER) then
cfgSafely(makefolder, CFG_FOLDER)
end
end

local function saveConfig(data)
ensureCfgFolder()
local ok, encoded = pcall(function() return HttpService:JSONEncode(data) end)
if ok then
cfgSafely(writefile, CFG_FOLDER.."/"..CFG_FILE..CFG_EXT, encoded)
end
end

local function loadConfig()
local path = CFG_FOLDER.."/"..CFG_FILE..CFG_EXT
if not cfgSafely(isfile, path) then return nil end
local content = cfgSafely(readfile, path)
if not content then return nil end
local ok, data = pcall(function() return HttpService:JSONDecode(content) end)
return ok and data or nil
end

    if record.obj and record.obj.Parent then
        for prop, cKey in pairs(record.tags) do
            if C[cKey] then
                if prop == "BackgroundColor3" and record.obj.BackgroundTransparency > 0.8 then
                    record.obj[prop] = C[cKey]
                else
                    record.obj[prop] = C[cKey]
                end
            end
        end
    end

local function Make(className, props)
local inst = Instance.new(className)
local tags = {}
for k, v in pairs(props) do
inst[k] = v
if typeof(v) == "Color3" then
for cKey, cVal in pairs(C) do
if v == cVal and cKey ~= "toggle_on" and cKey ~= "dot_red" and cKey ~= "dot_yel" and cKey ~= "dot_grn" then
tags[k] = cKey
break
end
end
end
end
if next(tags) then
table.insert(themeRegistry, {obj = inst, tags = tags})
end
return inst
end

local transparencyFrames = {}

uiTransparency = math.clamp(t, 0, 0.85)
for _, e in ipairs(transparencyFrames) do
if e.frame and e.frame.Parent then
local target = math.clamp(e.base + uiTransparency * (1 - e.base), 0, 0.90)
e.frame.BackgroundTransparency = target

if not corner then
local uc = Instance.new("UICorner")
uc.CornerRadius = e.radius or UDim.new(0, 16)
uc.Parent = e.frame
end
end
end
end

local SavedTools     = {}
local SavedWalkSpeed = 16
local SavedJumpPower = 50
local partState      = {}

local originalToggles = {}
local proxyToggles    = {}

local function trackCurrentState()
if not cfgSaveInventory then return end
local char = Players.LocalPlayer.Character
local bp   = Players.LocalPlayer:FindFirstChild("Backpack")

SavedTools = {}
if bp then
    for _, t in ipairs(bp:GetChildren()) do
        if t:IsA("Tool") then table.insert(SavedTools, t:Clone()) end
    end
end
if char then
    for _, t in ipairs(char:GetChildren()) do
        if t:IsA("Tool") then table.insert(SavedTools, t:Clone()) end
    end
    local hum = char:FindFirstChild("Humanoid")
    if hum then
        SavedWalkSpeed = hum.WalkSpeed
        SavedJumpPower = hum.JumpPower
    end
end


end

task.spawn(function()
while task.wait(2) do
trackCurrentState()
end
end)

local function applyTogglesToCharacter()
for uid, state in pairs(partState) do
if state then
local shouldRestore = true
if cfgSaveMode == "Favored" and not cfgFavorites[uid] then
shouldRestore = false
end

        if shouldRestore then
            -- Force a complete reinitialization of the feature logic by re-triggering the toggle logic
            if originalToggles[uid] and originalToggles[uid].toggle then
                originalToggles[uid].toggle.SetOn(true, false)
            end
        else
            -- Gracefully disable tools that shouldn't be restored
            if originalToggles[uid] and originalToggles[uid].toggle then
                originalToggles[uid].toggle.SetOn(false, false)
            end
        end
    end
end


end

Players.LocalPlayer.CharacterAdded:Connect(function(char)
if not cfgSaveInventory then return end
task.wait(1.5) -- Restore Delay System (waits for full load)

local hum = char:WaitForChild("Humanoid", 3)
if hum then
    hum.WalkSpeed = SavedWalkSpeed
    hum.JumpPower = SavedJumpPower
end

local bp = Players.LocalPlayer:WaitForChild("Backpack", 3)
if bp then
    for _, tClone in ipairs(SavedTools) do
        if not bp:FindFirstChild(tClone.Name) then
            tClone:Clone().Parent = bp
        end
    end
end

applyTogglesToCharacter()


end)

    partState[uid] = state
    
    if isProxy then
        if originalToggles[uid] then originalToggles[uid].toggle.SetOn(state, true) end
    else
        if proxyToggles[uid] then proxyToggles[uid].toggle.SetOn(state, true) end
    end
    
    -- Logic Executions
    if pName == "TPTool" then
        local plyr = Players.LocalPlayer
        if state then
            local bp = plyr:FindFirstChild("Backpack")
            local char = plyr.Character
            
            -- Cleanup any dead cloned tool instances from prior Save Inventory restorations
            if bp and bp:FindFirstChild("TPTool") then bp.TPTool:Destroy() end
            if char and char:FindFirstChild("TPTool") then char.TPTool:Destroy() end
            
            if bp then
                local t = Instance.new("Tool")
                t.Name = "TPTool"
                t.RequiresHandle = false
                t.Activated:Connect(function()
                    local mouse = plyr:GetMouse()
                    local c = plyr.Character
                    if c and c:FindFirstChild("HumanoidRootPart") then
                        c.HumanoidRootPart.CFrame = CFrame.new(mouse.Hit.Position + Vector3.new(0,3,0))
                    end
                end)
                t.Parent = bp
            end
        else
            local bp = plyr:FindFirstChild("Backpack")
            if bp and bp:FindFirstChild("TPTool") then bp.TPTool:Destroy() end
            local char = plyr.Character
            if char and char:FindFirstChild("TPTool") then char.TPTool:Destroy() end
        end

local function makeDraggableReal(object, handle)
local dragging = false
local relative = Vector2.zero
local insetOff = Vector2.zero
local sg = object:FindFirstAncestorWhichIsA("ScreenGui")
if sg and sg.IgnoreGuiInset then
local ok, inset = pcall(function() return GuiService:GetGuiInset() end)
if ok then insetOff = insetOff + inset end
end
handle.InputBegan:Connect(function(input, processed)
if processed then return end
if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
dragging = true
relative = Vector2.new(object.AbsolutePosition.X, object.AbsolutePosition.Y)
+ Vector2.new(object.AbsoluteSize.X * object.AnchorPoint.X, object.AbsoluteSize.Y * object.AnchorPoint.Y)
- UserInputService:GetMouseLocation()
end
end)
local ec = UserInputService.InputEnded:Connect(function(input)
if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
dragging = false
end
end)
local rc = RunService.RenderStepped:Connect(function()
if dragging then
local pos = UserInputService:GetMouseLocation() + relative + insetOff
object.Position = UDim2.fromOffset(pos.X, pos.Y)
end
end)
object.Destroying:Connect(function() ec:Disconnect(); rc:Disconnect() end)
end

SearchInput:GetPropertyChangedSignal("Text"):Connect(function()
local query = string.lower(SearchInput.Text)

local function filterList(togglesDict)
    for uid, data in pairs(togglesDict) do
        if data.pill then
            local searchName = data.name
            if not searchName and originalToggles[uid] then
                searchName = originalToggles[uid].name
            end
            
            if searchName then
                if query == "" or string.find(string.lower(searchName), query) then
                    data.pill.Visible = true
                else
                    data.pill.Visible = false
                end
            end
        end
    end
end

filterList(originalToggles)
filterList(proxyToggles)

local function updateSec(sec)
    local visibleCount = 0
    for _, child in ipairs(sec.Content:GetChildren()) do
        if child:IsA("Frame") and child.Visible then
            visibleCount = visibleCount + 1
        end
    end
    local CONTENT_H = visibleCount > 0 and ((PART_H + 5) * visibleCount) or 0
    sec.Content.Size = ud2(1,0,0,CONTENT_H)
    
    if sec.IsOpen() then
        sec.Wrapper.Size = ud2(1,-16,0, SECTION_H + 4 + CONTENT_H)
    end
    
    -- Dynamically adjust section visibility & handle auto-expanding based on results
    if query ~= "" then
        sec.Wrapper.Visible = (visibleCount > 0)
        if visibleCount > 0 and not sec.IsOpen() then
            sec.SetOpen(true)
        end
    else
        sec.Wrapper.Visible = true
    end
end

if favSectionObj then updateSec(favSectionObj) end
for _, tabList in pairs(tabSections) do
    for _, sec in ipairs(tabList) do
        updateSec(sec)
    end
end
updateCollapseAllVisibility()


end)

activeTab = tabName
for key, tab in pairs(tabs) do
local active = (key == tabName)
TweenService:Create(tab.Pill, tabTween, { Size = active and ud2(0,40,0,2) or ud2(0,0,0,2) }):Play()
TweenService:Create(tab.Button, tabTween, { TextColor3 = active and C.textPri or C.textSec, BackgroundTransparency = active and 0.2 or 0.5 }):Play()
end
homeContainer.Visible = (tabName == "Home")
k2k1Container.Visible = (tabName == "2K1")
k2Container.Visible   = (tabName == "K2")
updateCollapseAllVisibility()
end
