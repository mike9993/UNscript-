local CFG_FOLDER = "UNScripts"
local CFG_FILE   = "settings"
local CFG_EXT    = ".cfg"

local function cfgSafely(fn, ...)
if fn then
local ok, res = pcall(fn, ...)
if not ok then return nil end
return res
end
end

local function ensureCfgFolder()
if isfolder and not cfgSafely(isfolder, CFG_FOLDER) then
cfgSafely(makefolder, CFG_FOLDER)
end
end

local function saveConfig(data)
ensureCfgFolder()
local ok, encoded = pcall(function() return HttpService:JSONEncode(data) end)
if ok then
cfgSafely(writefile, CFG_FOLDER.."/"..CFG_FILE..CFG_EXT, encoded)
end
end

local function loadConfig()
local path = CFG_FOLDER.."/"..CFG_FILE..CFG_EXT
if not cfgSafely(isfile, path) then return nil end
local content = cfgSafely(readfile, path)
if not content then return nil end
local ok, data = pcall(function() return HttpService:JSONDecode(content) end)
return ok and data or nil
end

local function Make(className, props)
local inst = Instance.new(className)
local tags = {}
for k, v in pairs(props) do
inst[k] = v
if typeof(v) == "Color3" then
for cKey, cVal in pairs(C) do
                if v == cVal then
tags[k] = cKey
break
end
end
end
end
if next(tags) then
table.insert(themeRegistry, {obj = inst, tags = tags})
end
return inst
end

local SavedTools     = {}

local SavedJumpPower = 50

local proxyToggles    = {}

local function trackCurrentState()
if not cfgSaveInventory then return end
local char = LocalPlayer.Character
local bp   = LocalPlayer:FindFirstChild("Backpack")
SavedTools = {}
if bp then
for _, t in ipairs(bp:GetChildren()) do
if t:IsA("Tool") then table.insert(SavedTools, t:Clone()) end
end
end
if char then
for _, t in ipairs(char:GetChildren()) do
if t:IsA("Tool") then table.insert(SavedTools, t:Clone()) end
end
local hum = char:FindFirstChild("Humanoid")
if hum then
SavedWalkSpeed = hum.WalkSpeed
SavedJumpPower = hum.JumpPower
end
end
end

task.spawn(function()
while task.wait(2) do
trackCurrentState()
end
end)

local function applyTogglesToCharacter()
for uid, state in pairs(partState) do
if state then
local shouldRestore = true
if cfgSaveMode == "Favored" and not cfgFavorites[uid] then
shouldRestore = false
end
if shouldRestore then
if originalToggles[uid] and originalToggles[uid].toggle then
originalToggles[uid].toggle.SetOn(true, false)
end
else
if originalToggles[uid] and originalToggles[uid].toggle then
originalToggles[uid].toggle.SetOn(false, false)
end
end
end
end
end

LocalPlayer.CharacterAdded:Connect(function(char)
if not cfgSaveInventory then return end
task.wait(1.5)
local hum = char:WaitForChild("Humanoid", 3)
if hum then
hum.WalkSpeed = SavedWalkSpeed
hum.JumpPower = SavedJumpPower
end
local bp = LocalPlayer:WaitForChild("Backpack", 3)
if bp then
for _, tClone in ipairs(SavedTools) do
if not bp:FindFirstChild(tClone.Name) then
tClone:Clone().Parent = bp
end
end
end
applyTogglesToCharacter()
end)

local function runFullUninstall()
	UninstallOverlay.Visible = false
	pcall(function()
		local path = CFG_FOLDER.."/"..CFG_FILE..CFG_EXT
		if cfgSafely(isfile, path) then cfgSafely(delfile, path) end
	end)
	pcall(function()
		local cg = game:GetService("CoreGui")
		for _, v in ipairs(cg:GetChildren()) do
			if v.Name:find("UNS_ESP") then v:Destroy() end
		end
	end)
	espEnabled = false
	if Gui and Gui.Parent then Gui:Destroy() end
end

UninstallYesBtn.MouseButton1Click:Connect(runFullUninstall)
UninstallNoBtn.MouseButton1Click:Connect(function() UninstallOverlay.Visible = false end)

UserInputService.InputBegan:Connect(function(input, processed)
if not UninstallOverlay.Visible then return end
if input.KeyCode == Enum.KeyCode.Return or input.KeyCode == Enum.KeyCode.KeypadEnter then
runFullUninstall()
elseif input.KeyCode == Enum.KeyCode.Backspace then
UninstallOverlay.Visible = false
end
end)

-- ESP system
local function getPlayerESPColor(targetPlayer)
-- 1. Check default Roblox Teams Service
if targetPlayer.Team and typeof(targetPlayer.Team) == "Instance" and targetPlayer.Team:IsA("Team") then
return targetPlayer.Team.TeamColor.Color
end

-- 2. Check for common Custom Team Colors in the player object
local customColor = targetPlayer:FindFirstChild("TeamColor") or targetPlayer:FindFirstChild("PlayerColor")
if customColor then
if customColor:IsA("Color3Value") then
return customColor.Value
elseif customColor:IsA("BrickColorValue") then
return customColor.Value.Color
end
end

-- 3. Fallback Ally/Enemy Logic
if LocalPlayer.Team and targetPlayer.Team == LocalPlayer.Team then
return Color3.fromRGB(0, 255, 0) -- Green for Allies
end

-- 4. Default Fallback
return Color3.fromRGB(255, 0, 0) -- Red for Enemies/Unaffiliated
end

local function UpdateESP()
if not espEnabled then return end
for _, player in ipairs(Players:GetPlayers()) do
if player ~= LocalPlayer and not espObjects[player] then CreateESP(player) end
end
for player, data in pairs(espObjects) do
local char = player.Character; local hrp = char and char:FindFirstChild("HumanoidRootPart")
local hum = char and char:FindFirstChildOfClass("Humanoid"); local head = char and (char:FindFirstChild("Head") or hrp)
if char and hrp and head and hum and hum.Health > 0 then
local isEnemy = true
if espEnemyOnly and LocalPlayer.Team and player.Team then isEnemy = LocalPlayer.Team ~= player.Team end
if not isEnemy then
data.BillboardGui.Enabled = false; data.Highlight.Enabled = false
else
local espColor = getPlayerESPColor(player)
data.Highlight.Enabled = espChams; data.Highlight.Adornee = char; data.Highlight.FillColor = espColor
data.Highlight.OutlineColor = espColor
data.BillboardGui.Enabled = (espNames or espDistance); data.BillboardGui.Adornee = head
data.NameLabel.Visible = espNames; data.NameLabel.TextColor3 = espColor
local myHRP = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
if myHRP and espDistance then
local dist = (hrp.Position - myHRP.Position).Magnitude
data.DistLabel.Text = string.format("%d studs", dist); data.DistLabel.Visible = true; data.DistLabel.TextColor3 = espColor
else data.DistLabel.Visible = false end
end
else
data.BillboardGui.Enabled = false; data.Highlight.Enabled = false
end
end
end

local function lookupSliderParams(uid)

local function makeDraggableReal(object, handle)
local dragging = false
local relative = Vector2.zero
local insetOff = Vector2.zero
local sg = object:FindFirstAncestorWhichIsA("ScreenGui")
if sg and sg.IgnoreGuiInset then
local ok, inset = pcall(function() return GuiService:GetGuiInset() end)
if ok then insetOff = insetOff + inset end
end
handle.InputBegan:Connect(function(input, processed)
if processed then return end
if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
dragging = true
relative = Vector2.new(object.AbsolutePosition.X, object.AbsolutePosition.Y)
+ Vector2.new(object.AbsoluteSize.X * object.AnchorPoint.X, object.AbsoluteSize.Y * object.AnchorPoint.Y)
- UserInputService:GetMouseLocation()
end
end)
local ec = UserInputService.InputEnded:Connect(function(input)
if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
dragging = false
end
end)
local rc = RunService.RenderStepped:Connect(function()
if dragging then
local pos = UserInputService:GetMouseLocation() + relative + insetOff
object.Position = UDim2.fromOffset(pos.X, pos.Y)
end
end)
object.Destroying:Connect(function() ec:Disconnect(); rc:Disconnect() end)
end

for _, p in ipairs(Players:GetPlayers()) do
if p ~= LocalPlayer and p.Character and p.Character:FindFirstChild("Head") and p.Character:FindFirstChild("HumanoidRootPart") then
local head = p.Character.Head; local hum = p.Character:FindFirstChildOfClass("Humanoid")
if hum and hum.Health > 0 then
local screenPos, onScreen = Camera:WorldToViewportPoint(head.Position)
if onScreen then
local dist = (Vector2.new(screenPos.X, screenPos.Y) - Vector2.new(mousePos.X, mousePos.Y)).Magnitude
if dist < shortestDist then
local isVisible = true
if wallCheckEnabled then
local origin = Camera.CFrame.Position; local dest = head.Position
local result = workspace:Raycast(origin, dest - origin, rayParams)
if result and result.Instance and not result.Instance:IsDescendantOf(p.Character) then isVisible = false end
end
if isVisible then shortestDist = dist; closest = head end
end
end
end
end
end
if closest then Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, closest.Position), aimSmoothness) end
end
end)

local r = LocalPlayer.Character.HumanoidRootPart

end
end

end
